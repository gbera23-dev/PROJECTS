class Solution {
public:
    int findTargetSumWays(vector<int>& nums, int target, int index) {
        if(index == nums.size()) { 
            return target == 0; 
        }
        return findTargetSumWays(nums, target - nums[index], index + 1) + 
        findTargetSumWays(nums, target + nums[index], index + 1);
    }

    int findTargetSumWays(vector<int>& nums, int target) {
        return findTargetSumWays(nums, target, 0);
    }
};
